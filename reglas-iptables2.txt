#1Borrar reglas
iptables -F
iptables -Z
#0 Permitir SSH

IPTABLES -A INPUT -p tcp –sport 22 -s 172.22.x.x -j ACCEPT
iptables -A OUTPUT -p tcp –dport 22 -d 172.22.x.x -j ACCEPT

#2 Politica por defecto DROP
iptables -P INPUT DROP
iptables -P OUTPUT DROP
IPTABLES -P FORDWARD DROP

#5 permitir hacer ping pero no responder
iptables -A OUTPUT -p icmp --icmp-type echo-request -j ACCEPT
iptables -A INPUT -p icmp --icmp-type echo-reply -j ACCEPT

#7 HTTP, HTTPS
iptables -A OUTPUT -p tcp -m multiport --dports 80,443 -j ACCEPT
iptables -A INPUT -p tcp -m multiport --sports 80,443 -j ACCEPT

#8 Ahora permite ping de fuera (anula el 5)
iptables -D OUTPUT -p icmp --icmp-type echo-request -j ACCEPT
iptables -D INPUT -p icmp --icmp-type echo-reply -j ACCEPT

#9 Permite acceder a mi servidor apache a otros equipos
iptables -A OUTPUT -p tcp --sport 80 -j ACCEPT
iptables -A INPUT -p tcp --dport 80 -j ACCEPT
